function calculerTarif() { const resDiv = document.getElementById("resultat-tarif"); const errDiv = document.getElementById("erreur-tarif"); const flottant = document.getElementById("tarif-flottant"); const flottantTexte = document.getElementById("tarif-flottant-texte"); const depart = form["Départ"].value.trim(); const arrivee = form["Arrivee"].value.trim(); const trajetType = trajetSelect.value; const dateAllerValue = form["DateHeureAller"].value; const dateRetourValue = dateRetourInput.value; errDiv.style.display = "none"; errDiv.textContent = ""; resDiv.style.display = "none"; resDiv.textContent = ""; if (flottant) flottant.style.display = "none"; if (!depart || !arrivee) { errDiv.textContent = "Veuillez renseigner le départ et l'arrivée."; errDiv.style.display = "block"; form.scrollIntoView({ behavior: "smooth", block: "start" }); return; } if (!dateAllerValue) { errDiv.textContent = "Veuillez renseigner la date et heure de départ."; errDiv.style.display = "block"; return; } if (trajetType === "aller-retour" && !dateRetourValue) { errDiv.textContent = "Veuillez renseigner la date et heure de retour."; errDiv.style.display = "block"; return; } resDiv.style.display = "block"; resDiv.innerHTML = "Calcul en cours…"; distanceService.getDistanceMatrix( { origins: [depart], destinations: [arrivee], travelMode: google.maps.TravelMode.DRIVING, unitSystem: google.maps.UnitSystem.METRIC, }, (response, status) => { if (status !== "OK") { resDiv.style.display = "none"; errDiv.textContent = "Erreur Google Maps : " + status; errDiv.style.display = "block"; return; } const element = response.rows[0].elements[0]; if (element.status !== "OK") { resDiv.style.display = "none"; errDiv.textContent = "Adresse introuvable. Vérifiez l'orthographe."; errDiv.style.display = "block"; return; } const distanceMetres = element.distance.value; let distanceKm = Math.round((distanceMetres / 1000) * 10) / 10; const dureeTexte = element.duration.text; let tarif = BASE_TARIF; if (distanceKm > BASE_KM) { tarif += (distanceKm - BASE_KM) * SUPP_PAR_KM; } // Vérifier si tarif de nuit pour l'aller let heureAller = getHeureFromDateTime(dateAllerValue); if (heureAller >= 21 || heureAller < 7) { tarif = tarif * 1.20; } let distanceTotale = distanceKm; let tarifFinal = tarif; // Pour l'aller-retour if (trajetType === "aller-retour") { // Calcul du tarif retour let tarifRetour = BASE_TARIF; if (distanceKm > BASE_KM) { tarifRetour += (distanceKm - BASE_KM) * SUPP_PAR_KM; } let heureRetour = getHeureFromDateTime(dateRetourValue); if (heureRetour >= 21 || heureRetour < 7) { tarifRetour = tarifRetour * 1.20; } // Tarif total = aller + retour avec 10% de réduction sur le retour tarifFinal = tarif + (tarifRetour * 0.9); tarifFinal = Math.round(tarifFinal * 100) / 100; // Distance totale = aller x 2 distanceTotale = Math.round((distanceKm * 2) * 10) / 10; } else { tarifFinal = Math.round(tarif * 100) / 100; } derniereEstimation = { distanceKm, distanceTotale, dureeTexte, tarif: tarifFinal, trajetType }; let affichage = ""; if (trajetType === "aller-retour") { affichage = "Distance totale (A/R) : " + distanceTotale + " km • " + "Durée aller : " + dureeTexte + "
" + "Estimation Aller-Retour : " + tarifFinal.toFixed(2) + " €
" + "(réduction de 10% sur le trajet retour)"; } else { affichage = "Distance : " + distanceKm + " km • " + "Durée : " + dureeTexte + "
" + "Estimation Aller simple : " + tarifFinal.toFixed(2) + " €"; } resDiv.style.display = "block"; resDiv.innerHTML = affichage; if (flottant && flottantTexte) { if (trajetType === "aller-retour") { flottantTexte.textContent = "Aller-Retour : " + tarifFinal.toFixed(2) + " € • " + distanceTotale + " km (A/R)"; } else { flottantTexte.textContent = "Estimation : " + tarifFinal.toFixed(2) + " € • " + distanceKm + " km"; } flottant.style.display = "flex"; } // Affichage de la carte (trajet aller) if (directionsService && directionsRenderer) { directionsService.route( { origin: depart, destination: arrivee, travelMode: google.maps.TravelMode.DRIVING }, (result, statusDir) => { if (statusDir === "OK") directionsRenderer.setDirections(result); } ); } } ); }
